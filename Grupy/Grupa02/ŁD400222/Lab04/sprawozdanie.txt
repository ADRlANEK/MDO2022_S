dockerfile:

FROM ubuntu:latest AS git
RUN apt-get -y update
RUN apt-get -y install git
RUN git clone https://github.com/brandon1024/Java-Simple-Messenger.git


FROM git as java
RUN apt-get -y update
ARG DEBIAN_FRONTEND=noninteractive
ENV TZ=Europe/Moscow
RUN apt-get install -y tzdata


FROM java as app
RUN apt-get install -y default-jre && apt-get install -y default-jdk
RUN ./Java-Simple-Messenger/gradlew :server:run

dockerfile wykorzystuje obraz systemy linux ubuntu, update'uje apt oraz klonuje repozytorium z aplikacją.
Po pobraniu aplikacji ustawiana jest strefa czasowa, pobierane jest JRE oraz JDK oraz uruchamiany jest serwer aplikacji.
Aplikacja napisana jest w Javie, budowana jest przy użyciu Gradle.
