logvol none --grow --size=2048 --thinpool --metadatasize=104 --chunksize=64 --name=pool00 --vgname=fedora_fedora
logvol / --fstype="xfs" --grow --size=2048 --thin --poolname=pool00 --name=root --vgname=fedora_fedora
